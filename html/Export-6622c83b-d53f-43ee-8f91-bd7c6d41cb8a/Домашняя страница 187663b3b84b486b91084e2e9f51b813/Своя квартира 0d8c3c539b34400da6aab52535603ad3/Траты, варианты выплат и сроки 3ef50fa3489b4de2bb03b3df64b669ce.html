zpA = 140000 #ЗП Андрей
zpO = zpN + zpA #ЗП общая

#Подсчет ОБЯЗАТЕЛЬНЫХ трат
pit = 30000 #Продукты и вкусняшки
kom = 10000 #Коммуналка
otp1 = 15000 #Отпуск 1
otp2 = 15000 #Отпуск 2
tratN = 3000 + 1000 + 3000 + 4000 #Личные траты Насти (транспорт, интернет, подарки, нужды)
tratA = 3000 + 1000+ 4000 + 5000 #Личные траты Андрея (транспорт, интернет, подарки, нужды)
sumOT = pit + kom + otp1 + otp2 + tratN + tratA

#Сумма необходимая на ремонт
rem = 1500000
#Сумма необходимая на подушку безопасности
pod = 350000
#Сумма ипотеки
ipoteka = 15000000
#Минимальный ежемесячный взнос
ipotekaMIN = 70000 #минимально возможная сумма ипотеки (от 65 до 70к, откруглила в большую сторону, тк цены растут)

#План расходов 1 (2 отпуска в год)
svobod1 = zpO - sumOT #остается сумма в месяц
ost1 = svobod1 - ipotekaMIN #с этой суммы первые годы копим подушку и на ремонт
srok_pod1 = pod / ost1 #сколько месяцев копим на подушку
srok_rem1 = rem / ost1 #сколько месяцев копим на ремонт
#По такому плану копим подушку 7 месяцев и на ремонт почти 3 года. Только потом начинаем вкладываться в ипотеку (картинка 1)

#План расходов 2 (1 отпуск в год)
svobod2 = zpO - sumOT + otp2 #остается сумма в месяц
ost2 = svobod2 - ipotekaMIN #с этой суммы первые годы копим подушку и на ремонт
srok_pod2 = pod / ost2 #сколько месяцев копим на подушку
srok_rem2 = rem / ost2 #сколько месяцев копим на ремонт
#По такому плану копим на подушку 5 месяцев и на ремонт 2 года. Потом начинаем вкладываться в ипотеку (картинка 2)

#План расходов 3 (не ездим в отпуска)
svobod3 = zpO - sumOT + otp2 + otp1 #остается сумма в месяц
ost3 = svobod3 - ipotekaMIN #с этой суммы первые годы копим подушку и на ремонт
srok_pod3 = pod / ost3 #сколько месяцев копим на подушку
srok_rem3 = rem / ost3 #сколько месяцев копим на ремонт
#По такому плану копим на подушку 4 месяца и на ремонт 1,5 года. Потом начинаем вкладываться в ипотеку (картинка 3). Предлагаю его  и после 2 лет уже начинать ездить в отпуска

  • Архивные варианты

    Картинка 1. Начало ипотеки март 2023 - конец декабрь 2036 Ездим в отпуска, чилим без ремонта 3 года. Как только есть подушка и ремонт - вкладываемся в ипотеку с мая 2026

    Картинка 2. Начало ипотеки март 2023 - конец ноябрь 2034 Ездим в 1 отпуск, после накопления на все (2 года 4 месяца) начинаем вкладываться в ипотеку с июля 2025

    Картинка 3. Начало ипотеки март 2023 - конец август 2034 Не ездим в отпуска все время, что копим на ремонт и подушку (2 года), потом вкладываемся в ипотеку и 1 отпуск

    ДРУГОЙ ВАРИАНТ

    Картинка 4. Начало ипотеки март 2023 - конец июнь 2033 Не ездим вообще в отпуска, после того как копим на ремонт и подушку все скидываем в ипотеку

    Ответвление варианта 2

    Данный вариант компенсирует затраты на ребенка и обезопасит нас бОльшими накоплениями. После 2 лет накоплений на подушку и ремонт, у нас остается свободных денег (с учетом выплаты минималке по ипотеке) 63к. Если платить ежемесячно 50к досрочно, а 13к копить на ребенка, на траты, на туда-сюда, то получается так. + Это самый спокойный вариант и всегда отпуска
